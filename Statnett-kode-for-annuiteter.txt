for (current_year in y_NG_annuities) {

  ## Import av bruttoinvesteringer ####
  df_ig = read.xlsx("./Data/Statnett_Investment_stream/GROSS_Investment_stream.xlsx", sheet = as.character(y.cb))
  df_ig[is.na(df_ig)] = 0
  
  # Henter ut netto-oversikt over anleggene som var i drift og anskaffet i 1991 eller tidligere
  #### ENS 21.10.2025 - Det siste vilkÃ¥ret her fjerner reserveanlegg mÃ¥ avklares fÃ¸r varsel/vedtak ####
  df_NG_pre91 = df_NG_net[df_NG_net$cyr<=ISS & df_NG_net$ayr<=ISS & !is.na(df_NG_net$cyr),]
  # Setter avhendelsesÃ¥ret for alle anlegg som ikke er avhendet til 9999 for enkel logisk test.
  df_NG_pre91$dyr[is.na(df_NG_pre91$dyr)] <- 9999
  # Fjerner alle anlegg fra Ã¥pningsbalansen som er tatt ut av drift fÃ¸r Ã¥pningbalanse skal fordeles bakover.
  # Dette blir rett fordi verdien av anleggene ogsÃ¥ mÃ¥ tas ut pÃ¥ investeringssiden
  print(y.cb)
  df_NG_pre91 = df_NG_pre91[df_NG_pre91$dyr>y.cb,]
  
  #Beregner sum av NG capex per anleggskategori
  df_NG_pre91_grouped <- df_NG_pre91 %>%
    group_by(cyr, asset_type) %>%
    dplyr::summarise(sum_capex = sum(ng.capex), na.rm = TRUE) %>%
    ungroup()
  
  unique(df_NG_pre91_grouped$asset_type)
  
  df_NG_pre91_grouped$years_depreciated = ISS - df_NG_pre91_grouped$cyr
  
  asset_lifetimes <- c(
    "Linje" = lifetime_lines,
    "Kabel" = lifetime_cables,
    "Transformator" = lifetime_trafo,
    "HVDC-omformer" = lifetime_HVDC,
    "Avgang" = lifetime_ce,
    "Kompenseringsanlegg" = lifetime_cd
  )
  
  
  
  # Lager kolonne for Ã¥ lagre andel av 
  df_NG_pre91_grouped$share_capex_left_91 = 0
  
  # Regner andel av kapital som ikke er avskrevet i 1991
  df_NG_pre91_grouped$share_capex_left_91 = with(df_NG_pre91_grouped,

  depNGcapex_ce_pre91 = sum(df_NG_pre91_grouped$dep_NG_capex[df_NG_pre91_grouped$asset_type == "Avgang"])

  dep_HVDC_share_s_pre91 = depNGcapex_HVDC_pre91/depNGcapex_s_pre91

  df_NG_pre91_grouped_st = df_NG_pre91_grouped %>%

    ))


    
  # Regner ny sum av dep_NG_capex - MÃ¥ deflateres
  df_NG_pre91_grouped_st <- df_NG_pre91_grouped_st %>%
    group_by(cyr, asset_type) %>%
    dplyr::summarise(sum_dep_NG_capex = sum(dep_NG_capex), na.rm = TRUE) %>%
    ungroup()

  

  
  df_ssb_09181 <- as.data.frame(px_data_09181, column.name.type = "text", variable.value.type = "text")

  #rm(pxweb_query_list_11189)

  names(df_ssb_09181)[2]="prcent_change"

  df_ssb_08184 <- df_ssb_08184 %>%
    dplyr::mutate(prcent_change = c(0, (kpi[-1] - kpi[-n()]) / kpi[-n()] * 100))

## Deflaterer avskrevne NormGrid-verdier ####

  price_delta1991$year <- as.numeric(price_delta1991$year)

  df_NG_pre91_grouped_st$sum_dep_def_NG_capex <- NA

      df_NG_pre91_grouped_st$sum_dep_def_NG_capex[i] <- df_NG_pre91_grouped_st$sum_dep_NG_capex[i] / def_factor
    } else {
      stop(print("Found observations with no deflation factor."))
    }

  df_NG_pre91_grouped_st$pre_91_key = with(df_NG_pre91_grouped_st,

      asset_type == "Linje" ~ "Lines",
      asset_type == "Kabel" ~ "Cables",
      asset_type == "Stasjon" ~ "Substations",
      asset_type == "Kompenseringsanlegg" ~ "Compensatingdevices",
      TRUE ~ asset_alias  
    ))
  
  #Omstrukturerer slik at jeg fÃ¥r en kollonne pr anleggstype i investeringsstrÃ¸mmen
  df_NG_pre91_grouped_st <- df_NG_pre91_grouped_st %>%
    pivot_wider(
      names_from = asset_alias,

    select(cyr, everything())

  df_NG_pre91_grouped_st_result$estimated_asset_value_Lines <- with(df_NG_pre91_grouped_st_result,

  df_b = read.xlsx("./Data/Statnett_Investment_stream/BOUGHT_Investment_stream.xlsx", sheet = as.character(y.cb))

    mutate(
      Lines.ti_ann = Lines.in_ann + Lines.b_ann,
